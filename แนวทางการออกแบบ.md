การออกแบบ UI/UX สำหรับแอปพลิเคชันควบคุมอุปกรณ์ในร้านซ่อมมอเตอร์ไซค์ ต้องคำนึงถึงสภาพแวดล้อมหน้างานจริง (Contextual UX) เป็นหลัก ช่างซ่อมมักมือเปื้อนน้ำมัน ทำงานในที่แสงจ้าหรือมืดต่างกัน และต้องการความรวดเร็ว แม่นยำ ปลอดภัยในการสั่งงานอุปกรณ์แรงดันสูง เช่น ปั๊มลม หรือแม่แรงไฮโดรลิก🧠 1. หลักการออกแบบ UX/UI Core (สำหรับร้านซ่อมมอเตอร์ไซค์)โทนสี (Color Palette): แนะนำเป็น Dark Mode เป็นหลัก (เช่น สีพื้นหลัง #1E1E24 ดำ/เทาเข้ม) เพื่อซ่อนรอยนิ้วมือคราบน้ำมันบนจอ และใช้สีสว่างจัดจ้าสำหรับปั๊มลม เช่น สีส้ม (#FF9800) หรือ สีแดงอินดัสเทรียล (#E53935) เพื่อเตือนให้ระวังเรื่องความปลอดภัยเป้าหมายการกด (Touch Targets): ออกแบบปุ่มกดให้มีขนาดใหญ่เป็นพิเศษ (อย่างน้อย 64x64 พิกเซลสำหรับมือถือ) เผื่อช่างใส่ถุงมือหรือนิ้วเปื้อนระบบป้องกันการกดพลาด (Safety Confirmation): อุปกรณ์อันตราย เช่น ปั๊มลมขนาดใหญ่ ควรใช้ระบบ "กดค้าง 2 วินาที (Long Press)" หรือ "ปัดเพื่อเปิด (Slide to Turn On)" แทนการกดทีเดียว เพื่อป้องกันการเดินชนแล้วเปิดโดยไม่ได้ตั้งใจ📱 2. Layout แบบที่ 1: สำหรับมือถือ (Mobile layout)เน้นการถือใช้งานด้วยมือเดียวขณะที่มืออีกข้างกำลังซ่อมรถ การจัดเลย์เอาต์จะเป็นแนวตั้ง (Single Column) เรียงลำดับตามความสำคัญบนสุด (Status Bar): แสดงสถานะแรงดันลมปัจจุบัน (PSI) และอุณหภูมิของปั๊มลมเด่นชัดตรงกลาง (Large Action Buttons): ปุ่มเปิด-ปิดขนาดใหญ่ แยกสีสถานะชัดเจน (เขียว=เปิดทำงาน / แดง=ปิด)ล่างสุด (Bottom Navigation): เมนูสลับหน้าจอ (หน้าแรกควบคุม / หน้าตั้งค่าแรงดัน / หน้าประวัติการบำรุงรักษา)💻 3. Layout แบบที่ 2: สำหรับพีซี / แท็บเล็ตตั้งโต๊ะ (PC/Desktop Layout)เน้นการแสดงผลแบบ Dashboard สำหรับโต๊ะแคชเชียร์หรือหน้าเคาน์เตอร์รับรถ แสดงผลหลายคอลัมน์ (Multi-Column) เพื่อให้เห็นภาพรวมของร้านทั้งหมดในหน้าเดียวฝั่งซ้าย (Sidebar Menu): เมนูสำหรับเลือกโซนในร้าน (เช่น โซนซ่อมที่ 1, โซนซ่อมที่ 2, ห้องเครื่องมือหลัก)ตรงกลาง (Main Control Grid): แสดงการ์ดควบคุมอุปกรณ์แยกเป็นตาราง (Grid View) สามารถเปิด-ปิดปั๊มลม, ลิฟต์ยกรถ, หรือระบบระบายอากาศได้พร้อมๆ กันฝั่งขวา (Analytics & Alerts Panel): กราฟแสดงสถิติจำนวนชั่วโมงที่ปั๊มลมทำงาน, การแจ้งเตือนเมื่อถึงกำหนดเปลี่ยนถ่ายน้ำมันเครื่องปั๊มลม และระบบบันทึกเวลาทำงานของช่าง💻 4. ตัวอย่างโค้ด Flutter แบบ Responsive (แยก PC และมือถือ)ใน Flutter เราจะใช้ LayoutBuilder ควบคู่กับค่า Breakpoint (เช่น 600 พิกเซล) เพื่อตรวจสอบขนาดหน้าจอและสลับเลย์เอาต์โดยอัตโนมัติอย่างมีประสิทธิภาพ:dartimport 'package:flutter/material.dart';

class ShopControlPage extends StatelessWidget {
  const ShopControlPage({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: const Color(0xFF1E1E24), // พื้นหลัง Dark Mode สไตล์อู่ซ่อมรถ
      appBar: AppBar(
        title: const Text('ระบบควบคุมร้านซ่อมมอเตอร์ไซค์', style: TextStyle(color: Colors.white)),
        backgroundColor: const Color(0xFF121214),
        elevation: 0,
      ),
      // ใช้ LayoutBuilder ในการดึงขนาดหน้าจอของอุปกรณ์ในขณะนั้น
      body: LayoutBuilder(
        builder: (context, constraints) {
          if (constraints.maxWidth > 600) {
            // ถ้าหน้าจอกว้างกว่า 600px (เช่น PC, Mac, Tablet) ให้แสดงเลย์เอาต์แนวนอน
            return _buildDesktopLayout(context);
          } else {
            // ถ้าหน้าจอแคบกว่า 600px (เช่น โทรศัพท์มือถือ) ให้แสดงเลย์เอาต์แนวตั้ง
            return _buildMobileLayout(context);
          }
        },
      ),
    );
  }

  // 📱 เลย์เอาต์เวอร์ชันมือถือ (เน้นปุ่มใหญ่ ถือใช้งานมือเดียวง่าย)
  Widget _buildMobileLayout(BuildContext context) {
    return Padding(
      padding: const EdgeInsets.all(16.0),
      child: Column(
        children: [
          _buildPumpStatusCard(isFullWidth: true), // การ์ดแสดงแรงดันลมเด่นๆ ด้านบน
          const SizedBox(height: 20),
          Expanded(
            child: ListView(
              children: [
                _buildSafetySwitchCard('ปั๊มลมหลัก (Main Air Pump)', true),
                const SizedBox(height: 16),
                _buildSafetySwitchCard('ไฮโดรลิกยกรถ ช่อง 1', false),
                const SizedBox(height: 16),
                _buildSafetySwitchCard('ระบบระบายอากาศ', true),
              ],
            ),
          ),
        ],
      ),
    );
  }

  // 💻 เลย์เอาต์เวอร์ชัน Desktop/PC (แสดงผลแบบแดชบอร์ด มีคอลัมน์ซ้ายขวา)
  Widget _buildDesktopLayout(BuildContext context) {
    return Row(
      children: [
        // คอลัมน์ซ้าย: รายการควบคุมอุปกรณ์ทั้งหมดแบบ Grid
        Expanded(
          flex: 2,
          child: Padding(
            padding: const EdgeInsets.all(24.0),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                const Text('แผงควบคุมอุปกรณ์หน้างาน', 
                  style: TextStyle(color: Colors.white, fontSize: 22, fontWeight: FontWeight.bold)),
                const SizedBox(height: 20),
                Expanded(
                  child: GridView.count(
                    crossAxisCount: 2, // แบ่งเป็น 2 คอลัมน์ขนานกัน
                    crossAxisSpacing: 16,
                    mainAxisSpacing: 16,
                    childAspectRatio: 1.5,
                    children: [
                      _buildSafetySwitchCard('ปั๊มลมหลัก (Main Air Pump)', true),
                      _buildSafetySwitchCard('ไฮโดรลิกยกรถ ช่อง 1', false),
                      _buildSafetySwitchCard('ไฮโดรลิกยกรถ ช่อง 2', false),
                      _buildSafetySwitchCard('ระบบระบายอากาศ', true),
                    ],
                  ),
                ),
              ],
            ),
          ),
        ),
        // คอลัมน์ขวา: แสดงสถานะเชิงลึก (เช่น ค่า PSI ของปั๊มลม และ Alerts บำรุงรักษา)
        Expanded(
          flex: 1,
          child: Container(
            color: const Color(0xFF121214),
            padding: const EdgeInsets.all(24.0),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                const Text('สถานะระบบปัจจุบัน', 
                  style: TextStyle(color: Colors.white, fontSize: 18, fontWeight: FontWeight.bold)),
                const SizedBox(height: 20),
                _buildPumpStatusCard(isFullWidth: true),
                const SizedBox(height: 24),
                const Divider(color: Colors.grey),
                const SizedBox(height: 12),
                const Text('⚠️ บันทึกเตือนความปลอดภัย', style: TextStyle(color: Colors.amber, fontWeight: FontWeight.bold)),
                const SizedBox(height: 8),
                const Text('- ปั๊มลมทำงานต่อเนื่อง 4 ชั่วโมงแล้ว\n- กำหนดเปลี่ยนน้ำมันเครื่องปั๊มลม: อีก 5 วัน', 
                  style: TextStyle(color: Colors.white70, height: 1.5)),
              ],
            ),
          ),
        ),
      ],
    );
  }

  // Widget แสดงสถานะแรงดันลม (PSI)
  Widget _buildPumpStatusCard({required bool isFullWidth}) {
    return Container(
      width: isFullWidth ? double.infinity : null,
      padding: const EdgeInsets.all(24),
      decoration: BoxDecoration(
        color: const Color(0xFF2D2D34),
        borderRadius: BorderRadius.circular(16),
        border: Border.all(color: Colors.orange, width: 2), // ขอบสีส้มอุตสาหกรรม
      ),
      child: const Column(
        children: [
          Text('แรงดันลมปัจจุบัน', style: TextStyle(color: Colors.white70, fontSize: 16)),
          SizedBox(height: 8),
          Text('115 PSI', style: TextStyle(color: Colors.orange, fontSize: 48, fontWeight: FontWeight.bold)),
          Text('สถานะ: ปกติ (ตัดการทำงานอัตโนมัติเมื่อครบ 120 PSI)', style: TextStyle(color: Colors.green, fontSize: 12)),
        ],
      ),
    );
  }

  // Widget การ์ดสวิตช์เปิดปิดแบบปลอดภัย (ต้องกดค้างป้องกันการพลาด)
  Widget _buildSafetySwitchCard(String deviceName, bool isActive) {
    return Container(
      padding: const EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: const Color(0xFF2A2A30),
        borderRadius: BorderRadius.circular(12),
      ),
      child: Row(
        mainAxisAlignment: MainAxisAlignment.between,
        children: [
          Expanded(
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              mainAxisAlignment: MainAxisAlignment.center,
              children: [
                Text(deviceName, style: const TextStyle(color: Colors.white, fontSize: 16, fontWeight: FontWeight.bold)),
                const SizedBox(height: 4),
                Text(isActive ? 'กำลังทำงาน' : 'ปิดอยู่', 
                  style: TextStyle(color: isActive ? Colors.green : Colors.red, fontSize: 13)),
              ],
            ),
          ),
          // ปุ่มกดเปิดปิดขนาดใหญ่พิเศษ เหมาะกับงานช่าง
          ElevatedButton(
            style: ElevatedButton.styleFrom(
              backgroundColor: isActive ? Colors.red : Colors.green,
              minimumSize: const Size(80, 60), // ขนาดปุ่มใหญ่กดง่าย
              shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(8)),
            ),
            onPressed: () {
              // TODO: ใส่ Logic สำหรับเปิดปิด หรือใช้ GestureDetector ทำ Long Press เพื่อความปลอดภัย
            },
            child: Text(isActive ? 'STOP' : 'START', 
              style: const TextStyle(color: Colors.white, fontWeight: FontWeight.bold)),
          ),
        ],
      ),
    );
  }
}
